**Check 8 — GFC P/B compression: KOSPI**

def get_pb(df, country, year, month):
    # Find the month-end date for the given year-month
    mask = (
        (df["country"] == country)
        & (df["date"].dt.year == year)
        & (df["date"].dt.month == month)
    )
    rows = df[mask]
    if len(rows) == 0:
        return None
    return rows["pb"].iloc[0]

kospi_pre  = get_pb(df, "KOSPI",   2007, 10)
kospi_gfc  = get_pb(df, "KOSPI",   2008, 10)
